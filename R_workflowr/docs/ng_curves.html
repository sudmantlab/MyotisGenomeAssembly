angle_rotate <- function(angle){
    # subset1 <- "(angle < 90 | angle > 270)"
    # subset2 <- "(angle >= 90 & angle <=270)"
    if (angle < 90 | angle > 270){
      return(angle)
    } else {
      return(angle+180)
    }
}
h_readjust <- function(angle){
    if (angle < 90 | angle > 270){
      return(0)
    } else {
      return(1)
    }
}
